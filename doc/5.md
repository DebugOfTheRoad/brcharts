####2.option示例
```js
var option={
    containerId:"container", //放置HighCharts容器对应的ID
    shows:["报警次数","网络攻击次数"],         //要显示的指标
    title:"散点图",          //标题
    subtitle:"副标题",       //副标题
    yTitle:"次数",         //y轴标题
    yUnit:"次" ,             //y轴单位
    xName:"datetime"       //x轴显示的名称
};
```
